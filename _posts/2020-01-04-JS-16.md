### 유효범위(Scope)

```js
var vscope = 'global';
function fscope(){
    alert(vscope);      // global
}
fscope();
```

```js
var vscope = 'global';
function fscope(){
    var vscope = 'local';
    alert('함수안 '+vscope);        // local
}
fscope();
alert('함수밖 '+vscope);            // global
```

### 유효범위의 효용

```js
function a (){
    var i = 0;
}
for(var i = 0; i < 5; i++){
    a();
    document.write(i);
}
// 01234
```

```js
function a (){
    i = 0;
}
for(i = 0; i < 5; i++){
    a();
    document.write(i);
}
// 00000
```
